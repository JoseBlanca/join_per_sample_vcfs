def generate_reference(group:&GroupedVariants) -> String:
    // A reference sequence should be generated for each sample and once all are generated we have to check that they are all exactly the same. If they are not we have to return a NonMatchingReferences error.
    reference = Option<String>
    for sample in get_samples_from_group(group):
        sample_reference = generate_reference_for_sample(&group, &sample)
        match reference {
            None => reference = sample_reference
            Some(reference) => if reference != sample_reference {return Err(NonMatchingReferences);} else {reference}
            
        }
    return reference

    variant_regions = Vec<(group.start, group.end)>
